103inline i64_ll qpow(i64_ll a, i64_ll b, i64_ll mod) {
104 assert(b >= 0);
105 i128_ll ans = 1;
106 while (b) {
107 if (b & 1)
108 ans = 1ll * ans * a % mod;
109 a = 1ll * a * a % mod;
110 b >>= 1;
111 }
112 return ans;
113}

120inline bool is_prime(i64_ll n) {
121 if (n < 3 || n % 2 == 0)
122 return n == 2;
123 i64_ll u = n - 1, t = 0;
124 while (u % 2 == 0)
125 u /= 2, ++t;
126 i64_ll ud[] = {2, 325, 9375, 28178, 450775, 9780504, 1795265022};
127 for (i64_ll a : ud) {
128 i128_ll v = qpow(a, u, n);
129 if (v == 1 || v == n - 1 || v == 0)
130 continue;
131 for (int j = 1; j <= t; j++) {
132 v = v * v % n;
133 if (v == n - 1 && j != t) {
134 v = 1;
135 break;
136 }
137 if (v == 1)
138 return 0;
139 }
140 if (v != 1)
141 return 0;
142 }
143 return 1;
144}
